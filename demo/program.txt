// testing loops
LET('i', 0)
REPEAT(5) {
  SET ('i', ADD(i, 1))
  PRINT(ADD('Count: ', i))
}

// testing function being
// used before its declaration
GREET('World')
TEST()

// testing local scoping
FUNCTION('TEST') {
  GREET()
  FUNCTION('GREET') {
    PRINT('This is the local GREET function')
  }
}

FUNCTION('GREET', 'name') {
  PRINT(ADD(ADD('Hello ', name), '!'))
}
